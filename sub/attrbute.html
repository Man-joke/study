바닐라 자바스크립트로 data- 값 가져오기

h2에 텍스트 활성화

img요소 src 값 넣기

img요소 alt 값 넣기

<script>

let item = document.querySelectorAll(".item");

let imgTitle = document.querySelector(".img-box h2");

let img = document.querySelector(".img-box img");


item.forEach(element => {

element.addEventListener("mouseenter", function(e){

let a = this.innerText;

imgTitle.innerText = a;


let b = this.childNodes;

let c = b[1].childNodes

let d = c[1].getAttribute('data-name')


img.alt = a;

img.src = "../images/" + d + ".png";

})

});

</script>
